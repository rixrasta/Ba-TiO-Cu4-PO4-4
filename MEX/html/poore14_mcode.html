1% pöördemaatriksi ehitamisega euleri nurkade a,b,g jaoks
 2
 3function [R]=poore(a,b,g)
 4
 5pir=pi/180;
 6
 7ag=a*pir;
 8bg=b*pir;
 9gg=g*pir;
10
11    Rg11=cos(ag)*cos(bg)*cos(gg)-sin(ag)*sin(gg);
12    Rg12=sin(ag)*cos(bg)*cos(gg)+cos(ag)*sin(gg);
13    Rg13=-sin(bg)*cos(gg);
14   Rg21=-cos(ag)*cos(bg)*sin(gg)-sin(ag)*cos(gg);
15   Rg22=-sin(ag)*cos(bg)*sin(gg)+cos(ag)*cos(gg);
16   Rg23=sin(bg)*sin(gg);
17   Rg31=cos(ag)*sin(bg);
18   Rg32=sin(ag)*sin(bg);
19   Rg33=cos(bg);
20   R=[Rg11 Rg12 Rg13
21      Rg21 Rg22 Rg23
22      Rg31 Rg32 Rg33];
