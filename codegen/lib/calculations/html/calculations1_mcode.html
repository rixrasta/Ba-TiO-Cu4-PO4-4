1function F = calculations(a1,a2,a3,B1) %#codegen
 2    %%Finds a list of intentsities corresponding to the provided angles
 3        %Arguments: a1,a2,a3 - euler angles; B1 - rotation matrix
 4        %Return: List of values. 
 5        
 6    Szz=1.952e-3;
 7    Syy=1.832e-3;
 8    Sxx=1.298e-3;
 9    w0=80.986e6;  %31P jaoks 600MHz väljas [Hz]
10    Shift=[1+Sxx 0 0;0 1+Syy 0; 0 0 1+Szz];     %nihke tensor PAS-s
11    A1=poore(a1,a2,a3);%a1k(m),a2k(m),a3k(m));
12    x=(-12:372);
13    n=length(x);
14    %A1i=inv(A1);
15    
16    %5/2 tuuma spinni operaatorid
17    Ix=[0 1; 1 0]/2;
18    Iy=[0 1; -1 0]/(2*i);
19    Iz=[1 0; 0 -1]/2;
20    %mõõteoperaator:
21    Ip=2*Ix;
22    
23    
24    %nivoode vahe
25    f12=zeros(1,n);
26    %intensiivsus
27    I12=f12;
28
29    freq1=f12;
30    g12=zeros(n);
31    c3=0;
32    c2=90;
33    for k=1:n
34        c1=x(k);
35        C1=poore(c1,c2,c3);
36        %C1i=inv(C1);
37        T1=C1*B1*A1*Shift/A1/B1/C1;
38        freq1(k)=T1(3,3);
39        Hz1=w0*(T1(3,1)*Ix + T1(3,2)*Iy + T1(3,3)*Iz);
40
41        [X1,E1]=eig(Hz1); %X on maatriks omavektoritest ,E on omavaartused
42        E=real(diag(E1));
43        U=X1\Ip*X1;
44        U2=abs(U) .^2;
45
46        %sagedus
47        g12(k)=E(2)-E(1);
48        f12(k)=(g12(k)-w0)/w0*1e2;
49
50        %intensiivsus
51        I12(k)=U2(1,2);
52    end
53    F=zeros(1,n);
54    F(1,:)=f12; %F(1,:)
55end
